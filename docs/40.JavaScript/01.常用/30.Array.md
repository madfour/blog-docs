## 数组去重
```js
/**
 * @param {Array}
 */
export function unique(arr) {
  if (!Array.isArray(arr)) {
    throw new Error('The first parameter must be an array')
  }
  if (arr.length == 1) {
    return arr
  }
  return [...new Set(arr)]
}
// unique([1,1,1,1,1])//[1]
```
